#实现一个Promise功能

  1. Promise 主要是为了解决异步回调的问题

  2. 封装一个简单的Promise

    • 初步构建

      function Promise(fn) {
      var callback;
        //实例方法then，用来注册异步事件
        this.then = function(done) {
            callback = done;
        }

        function resolve() {
            callback();
        }
        // 把内部定义resolve方法作为参数传到fn中，调用fn
        fn(resolve)         
     }

    • 链式调用

       function Promise(fn) {
    // 链式，成功回调的方法得变成数组才能存储
    var promise = this,
        value = null,
        promise._resolves = [];

    this.then = function(onFulfilled) {
           promise._resolves.push(onFulfilled);
        return this;
      }
     //resolve方法接收一个参数，即异步操作返回的结果，方便传值。
    function resolve(value) {
        // 保证resolve 执行过程是异步操作
        setTimeout(function() {
            promise._resolves.forEach(function(callback) {
               callback(value);
            });
        }, 0);
    }
    fn(resolve)
}

    • 引入状态

       function Promise(fn) {
      // 链式，成功回调的方法得变成数组才能存储
       var promise = this,
           value = null,
           promise._resolves = [],
           promise._status = 'PENDING';

    this.then = function(onFulfilled) {
        if (promise._status == 'PENDING') {
            promise._resolves.push(onFulfilled);
            return this;
        }
        onFulfilled(value);
        return this;
    }

    // resolve方法接收一个参数，即异步操作返回的结果，方便传值。
    function resolve(value) {
        setTimeout(function() {
            promise._status = "FULFILLED";
            promise._resolves.forEach(function(callback) {
                callback(value);
            });
        }, 0);
    }
    fn(resolve)
}
