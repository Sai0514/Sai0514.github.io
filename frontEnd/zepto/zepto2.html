uniq = function(array){
    return filter.call(array, function(item, idx){
        // 匹配item在数组中第一次出现的位置是否与item的位置相同
        return array.indexOf(item) == idx
    })

camelize, // 破折号分隔的字符串转换成驼峰形式
camelize = function(str){
    return str.replace(/-+(.)?/g, function(match, chr){
        //(.) 捕获任意非\n的单字符
        return chr?chr.toUpperCase():’’
    })
}
function dasherize(str) { // camelize的逆过程
    // 分为四个过程，分别替换字符来一步一步实现转换效果
    return str.replace(/::/g, ‘/‘)
           // (连续出现一次或多次的大写字母)(连续出现一个大写字母和一个小写字母);
           .replace(/([A-Z]+)([A-Z][a-z])/g, ‘$1$2’)
           //（一次小写字母或者一次数字）（一次大写字母）
           .replace(/([a-z\d])([A-Z])/g, ‘$1$2’)
           .replace(/_/g, ‘-‘)
           .toLowerCase()
}

分析—类数组

  1. 首先，公司的代码用的还是去年的版本，只是判断了 对象的长度是数字
  2. 新版本做了更多的限制判断：先看一下浏览器输出的arguments

所以：类数组满足以下要求

  • 不是function、window对象
  • array属于类数组、length===0(空数组)、length是不为0的数字且obj[length-1]属于obj
