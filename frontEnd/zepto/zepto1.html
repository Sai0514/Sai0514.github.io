Zepto源码解读之框架结构

var zepto = {}, $
function Z(dom,selector){}
zepto.Z = function(dom, selector) {
    return new Z(dom, selector)
}
zepto.isZ = function(object) {
    return object instanceof zepto.Z
}
zepto.init = function (selector, context) {
    return zepto.Z(dom, selector)
}
$ = function(selector, context){
    return zepto.init(selector, context)
}
$.fn = {}
zepto.Z.prototype = Z.prototype = $.fn
return $


// If `$` is not yet defined, point it to `Zepto`
window.Zepto = Zepto
window.$ === undefined && (window.$ = Zepto)

前情提要：

// zepto 不支持 CommonJs 规范的方法。项目中为了实现异步加载的方式，使用module.exports导出模块。即module.exports = Zepto; ajax，form等使用的话需要先加载依赖【var Zepto = require(‘./zepto.js’)】

如果使用webpack构建工具来进行异步加载的话，可以参考这篇文章 https://sebastianblade.com/how-to-import-unmodular-library-like-zepto/

框架结构：

  1. 源码中首先定义了zepto闭包函数，闭包函数中，声明$这个对象，并挂载了属性和方法（其自定义的扩展方法使用$.fn) 。最后返回$对象。因为该闭包函数是一个IIFE，即加载完毕后立即执行(实际上执行的是$.init)。最后，将zepto赋值全局对象window.$和window.zepto
  2. 从整体框架上来看，zepto对象首先执行init函数。可以看到init()入参是选择器和上下文，出参是调用Z()方法的返回值；Z()入参是dom元素和选择器，出现时Z对象的实例。因此，init返回的实际上是Z的实例。
  3. $返回的是Z的实例，因此展开可以得到dom的属性，但是如果想使用dom操作方法的话，zepto实例上是没有任何操作的。因此我们只能把$上的方法赋给zepto的原型上，这样在执行的时候就可以使用zepto调用实例方法。$上的方法都是挂载在$.fn = {}上。故，zepto.Z.prototype = Z.prototype = $.fn
  4. 解释一下，为啥会有zepto.Z.prototype = Z.prototype？这里看到了一篇博客，写的很详细。https://zhuanlan.zhihu.com/p/26185164
    我理解这里有点类似于考虑到递归的问题，例如: 传过来的值本身就是zepto.Z的实例，那么就将$.fn赋值给zepto.Z实例的原型上。

学习一下init()判断选择器,返回dom元素集合


  zepto.init = function(selector, context) {
    var dom
    // 如果selector为空，则返回$()
    if (!selector) return zepto.Z()
    // 如果selector为字符串： fragmentRE 正则判断
    else if (typeof selector == ‘string’) {
      selector = selector.trim()
      if (selector[0] == ‘<’ && fragmentRE.test(selector))
        // fragment 返回的是“由html字符串及标签名”组成的html片段 生成的dom节点集合
        dom = zepto.fragment(selector, RegExp.$1, context), selector = null
      // find 通过调用find函数寻找选择器dom
      else if (context !== undefined) return $(context).find(selector)
      // qsa 通过querySelectorAll寻找dom
      else dom = zepto.qsa(document, selector)
    }
    // 如果selector为函数： 等dom加载完毕再执行函数
    else if (isFunction(selector)) return $(document).ready(selector)
    // 如果selector为zepto对象，则直接返回
    else if (zepto.isZ(selector)) return selector
    else {
      // 如果selector为数组，则dom为其扁平化返回值。。。从这句代码可以看出，实际上dom输出的就是数组集合
      if (isArray(selector)) dom = compact(selector)
      // 如果selector为对象，返回为数组结构
      else if (isObject(selector))
        dom = [selector], selector = null
      else if (fragmentRE.test(selector))
        dom = zepto.fragment(selector.trim(), RegExp.$1, context), selector = null
      else if (context !== undefined) return $(context).find(selector)
      else dom = zepto.qsa(document, selector)
    }
    return zepto.Z(dom, selector)
  }

看完源码，基本可以明白init()返回的dom元素集合是数组结构。主要有以下几种判断：

  • (1) selector 为空，返回$
  • (2) selector 为字符串[$(‘p’),$(‘<div>’),$(‘#content’)等等]
  • (3) selector 为函数
  • (4) selector 为zepto对象[$(‘$(‘p)’)]
  • (5) selector 为其他类型[数组？对象？qsa万能查找？]
